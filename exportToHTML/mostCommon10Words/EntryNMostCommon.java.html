8    public class EntryNMostCommon

10     // searches for n most common words in text 
11     public static void printNMostCommonWords(String text, int n) 
12     { 
13       String[] words = text.split(",*\\s+"); 
14    
15        HashMap<String,Integer> occurences = new HashMap<String, Integer>(); 
16        for (String w:words){ 
17          if ( occurences.containsKey(w)) 
18            occurences.put(w,occurences.get(w)+1); 
19          else 
20            occurences.put(w,1); 
21        } 
22       TreeSet<Map.Entry<String, Integer>> occurSorted = new TreeSet(new Comparator<Map.Entry<String, Integer>>() 
23        { 
24          @Override 
25          public int compare(Map.Entry<String, Integer> o1, Map.Entry<String, Integer> o2) 
26          { 
27            return - o1.getValue().compareTo(o2.getValue()); 
28          } 
29        }); 
30    
31    
32        for (Map.Entry<String,Integer> entry: occurences.entrySet()) 
33        { 
34          occurSorted.add(new AbstractMap.SimpleEntry<String,Integer>(entry.getKey(),entry.getValue())); 
35        } 
36    
37       int i=0; 
38       for (AbstractMap.Entry<String,Integer> e:occurSorted) 
39       { 
40         if (i < n) 
41           System.out.println(e.getKey()); 
42         else 
43           break; 
44         i++; 
45       } 
46     } 
47     public static void main(String[] args) 
48     { 
49       String text = "abc def abc abc, ghk, def, 12, 34, 56 12 7 7 7 7 65 65 65 65 65"; 
50       printNMostCommonWords(text,5);
